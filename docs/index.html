cache233-go

High-performance in-memory caching library for Go

Go 1.24+ Apache 2.0

Features

W-TinyLFU Eviction

Adaptive eviction policy with high hit rates across all workload types.

High Throughput

Excellent performance under high contention with lock-free reads.

Low Memory

Minimal memory overhead with code-generated node types.

Quick Start

package main

import (
    "context"
    "time"

    "github.com/neko233-com/cache233-go"
    "github.com/neko233-com/cache233-go/stats"
)

func main() {
    ctx := context.Background()
    counter := stats.NewCounter()

    cache := cache233.Must(&cache233.Options[string, string]{
        MaximumSize:       10_000,
        ExpiryCalculator:  cache233.ExpiryAccessing[string, string](time.Minute),
        StatsRecorder:     counter,
    })

    cache.Set("user:1", "Alice")
    if val, ok := cache.GetIfPresent("user:1"); ok {
        fmt.Println(val) // Alice
    }
}

API Reference

MethodDescription
Must(o) / New(o)Create cache from Options
Get(ctx, key, loader)Get with auto-loading via SingleFlight
GetIfPresent(key)Get without side effects
Set(key, value)Insert or update
Compute(key, fn)Atomic compute operation
Invalidate(key)Remove single entry
InvalidateAll()Clear entire cache
CleanUp()Perform maintenance
EstimatedSize()Approximate entry count
Stats()Get statistics snapshot

Configuration

cache := cache233.Must(&cache233.Options[string, MyValue]{
    MaximumSize:      10_000,           // Max entries
    ExpiryCalculator: cache233.ExpiryAccessing[string, MyValue](5 * time.Minute),
    RefreshCalculator: cache233.RefreshWriting[string, MyValue](time.Minute),
    OnDeletion: func(e cache233.DeletionEvent[string, MyValue]) {
        log.Printf("evicted: %s (cause: %s)", e.Key, e.Cause)
    },
    StatsRecorder: stats.NewCounter(),
})
